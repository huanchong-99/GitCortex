    f: impl Fn() -> impl Future<Output = anyhow::Result<T>>,
) -> anyhow::Result<T> {
    let max_retries = 3;
    for attempt in 0..max_retries {
        match f().await {
            Ok(result) => return Ok(result),
            Err(e) if attempt < max_retries - 1 => {
                tokio::time::sleep(Duration::from_millis(1000 * (attempt + 1) as u64)).await;
            }
            Err(e) => return Err(e),
        }
    }

}
